/**
 * Extrai colunas específicas de uma aba da planilha
 * @param {GoogleAppsScript.Spreadsheet.Sheet} aba 
 * @param {string[]} colunas 
 * @returns {Object} Objeto contendo os dados extraídos
 */
function extrairColunas(aba, colunas) {
  var matrizDados = aba.getDataRange().getValues();
  var colunasIndices = colunas.map(col => col.charCodeAt(0) - 65); // Converte "A" -> 0, "C" -> 2, etc.

  var resultado = matrizDados.map(linha => 
    colunasIndices.map(i => (linha[i] !== "" ? linha[i] : 0)) // Se estiver vazio, retorna 0
  );

  return { dados: resultado };
}

/**
 * Utilizada pelo robô de consulta na caixa
 * Atualiza a aba "3PP" em um arquivo externo com base nos resultados de uma aba local "resultado pp".
 * A função extrai o ID da planilha externa de uma fórmula IMPORTRANGE, mapeia os dados de "resultado pp" e os usa para
 * atualizar as colunas "Contas", "Saldo Total", o mês específico, "Status" (marcando erros) e "Consulta" na planilha "3PP".
 * Após a atualização, a aba "resultado pp" é limpa.
 * @param {string} mes - O nome do mês (ex: "JAN", "FEV") cuja coluna de valor pago deve ser atualizada na planilha "3PP".
 * @param {string[]} erros - Um array de IDs de processo (limpos de pontuações) que devem ser marcados com "Erro" na coluna "Status" da planilha "3PP".
 * @param {(string|Date)} dataValor - A data da consulta ou atualização a ser registrada na coluna "Consulta" da planilha "3PP". Pode ser uma string de data ou um objeto Date.
 * @returns {void}
 */
function atualizarPP(mes, erros, dataValor) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var abaResultado = ss.getSheetByName("resultado pp");

  var formula = ss.getSheetByName("PP").getRange("A1").getFormula();
  var match = formula.match(/IMPORTRANGE\("https:\/\/docs\.google\.com\/spreadsheets\/d\/([a-zA-Z0-9-_]+)\//);

  if (!match || !match[1]) {
    Logger.log("Erro: Não foi possível extrair o ID da fórmula.");
    return;
  }

  var planilhaId = match[1];
  var planilhaExterna = SpreadsheetApp.openById(planilhaId);
  var abaPP = planilhaExterna.getSheetByName("3PP");

  if (!abaResultado || !abaPP) {
    Logger.log("Erro: Abas não encontradas.");
    return;
  }

  var dadosResultado = abaResultado.getDataRange().getValues();
  var dadosPP = abaPP.getDataRange().getValues();

  var header = dadosPP[0];
  var colContas = header.findIndex(col => col.toString().trim().toLowerCase() === "contas");
  var colSaldoTotal = header.findIndex(col => col.toString().trim().toLowerCase() === "saldo total");
  var colMes = header.findIndex(col => col.toString().trim().toLowerCase() === mes.toLowerCase());
  var colStatus = header.findIndex(col => col.toString().trim().toLowerCase() === "status");
  var colData = header.findIndex(col => col.toString().trim().toLowerCase() === "consulta" );
  if (colContas === -1 || colSaldoTotal === -1 || colMes === -1) {
    Logger.log("Erro: Coluna 'Contas', 'Saldo Total' ou '" + mes + "' não encontrada.");
    return;
  }
  
  //mapeia os dados da PP 
  var mapaResultado = {};
  for (var i = 1; i < dadosResultado.length; i++) {
    var processo = dadosResultado[i][0];
  

    if (processo) {
      mapaResultado[processo.replace(/[.\-]/g, "")] = {
        contas: dadosResultado[i][1],
        saldo_novo: dadosResultado[i][3],
        valor_pago: dadosResultado[i][4]
      };
    }
  }

  for (var j = 1; j < dadosPP.length; j++) {
    var processoPP = dadosPP[j][0];
    if (processoPP) {
      var processoLimpo = processoPP.replace(/[.\-]/g, "");

      if (erros.includes(processoLimpo) && colStatus !== -1) {
          abaPP.getRange(j + 1, colStatus + 1).setValue("Erro");
      }
      var dados = mapaResultado[processoLimpo];
   
      if (dados) {
        // Atualiza 'Contas'
        abaPP.getRange(j + 1, colContas + 1).setValue(dados.contas);
        // Atualiza 'Saldo Total'
        abaPP.getRange(j + 1, colSaldoTotal + 1).setValue(dados.saldo_novo);
        abaPP.getRange(j + 1, colStatus + 1).setValue("");
        // Atualiza coluna do mês
        var celulaMes = abaPP.getRange(j + 1, colMes + 1);
        var valorAtual = celulaMes.getValue();
        var novoValor = Number(dados.valor_pago) || 0;
        // soma se o valor anterior for numérico e diferente de zero
        var valorSomado = 0;
        if (valorAtual !== "" && !isNaN(valorAtual)) {
          valorSomado = Number(valorAtual);
        }
        if (colData !== -1) {
          const dataFinal = new Date(dataValor);
          if (!isNaN(dataFinal)) {
            abaPP.getRange(j + 1, colData + 1).setValue(dataFinal);
           ;

          } else {
            Logger.log("Data inválida recebida: " + dataValor);
          }
        }

        celulaMes.setValue(valorSomado + novoValor);
      }
    }
  }

  abaResultado.getDataRange().clearContent();
  Logger.log("Atualização concluída!");
}
