/**
 * Função responsável por enviar os dados da planilha para a página 'Execução de penas' do site
 */
function enviarDadosParaAnalise() {
  const PSC = SpreadsheetApp.openByUrl("https://docs.google.com/spreadsheets/d/1OpwOjfYiqPXH0324DGbYDHf9syDlPcS0uIBB771BmmY/edit");
  const PP = SpreadsheetApp.openByUrl("https://docs.google.com/spreadsheets/d/1ouGkmzF8_dJUZPXRigsL9qpgZ9hMiBOtaqbZs3utxNU/edit");
  const abaDadosPP = PP.getSheetByName('Dados');
  const abaDadosPSC = PSC.getSheetByName('Dados');
  //ATENÇÃO: os gráficos dessa página não crescem dinamicamente conforme as tabelas,
  //isso só existe no 'Processos da vara', lógica pode ser implementada no futuro
  const dadosPP = abaDadosPP.getRange(7, 2, 15, 3).getValues();
  const dadosPSC = abaDadosPSC.getRange(2, 1, 25, 3).getValues();
  let retorno = {
    PSC : montarObjeto(dadosPSC, 'PSC'),
    PP : montarObjeto(dadosPP, 'PP')
  }
  //construção do objeto json 
  return ContentService.createTextOutput(JSON.stringify(retorno))
  .setMimeType(ContentService.MimeType.JSON);
}

/**
 * resgata os dados da matriz e os transforma em um objeto
 * @param {string[][]} dados
 * @param {string} tipo deve ser 'PSC' ou 'PP'
 * @return {object} 
 */
function montarObjeto(dados, tipo) {
  let resultado = {};
  let atual = ""
  for (let i = 0; i < dados.length; i++) {
    if (dados[i][0] != "") {
      atual = dados[i][0].replace(/ /g, "_");
      if (!resultado[atual])
        resultado[atual] = {};
      if (tipo === 'PSC')
        resultado[atual][dados[i][1].replace(/ /g, "_")] = dados[i][2];
      else
        resultado[atual] = dados[i][1];
    }
    else if (dados[i][1] != "") {
      resultado[atual][dados[i][1].replace(/ /g, "_")] = dados[i][2];
    }
  }
  
  return resultado;
}

/**
 * Função responsável por buscar na planilha destinação de valores e enviar para a aba dados 4º vara do site,
 * feita para ser robusta caso ocorram mudanças na planilha, como a adição de novas abas ou colunas
 */
function destinacaoValores() {
  const planilha = SpreadsheetApp.openByUrl("https://docs.google.com/spreadsheets/d/1iFrRNHH_kw8mhn45QUFXKHyltehO3ILTAWO7XVx6MeY/edit?gid=1585728098#gid=1585728098");
  const anoAtual = new Date().getFullYear();
  //busca apenas abas referentes a anos válidos
  const abas = planilha.getSheets().filter(function (sheet) {
      const nomeAbaNum = parseInt(sheet.getName());
      // Verifica se o nome da aba é um número válido e representa um ano
      return !isNaN(nomeAbaNum) && nomeAbaNum > 1900 && nomeAbaNum <= anoAtual; // Assumindo anos válidos a partir de 1900
    });
    
    let objetoFinal = {};
    //inicializa as classes mais externas
    abas.forEach(function (aba) {
      const nomeAba = aba.getName();
      objetoFinal[nomeAba] = [];
      const categorias = buscarCategorias(aba).map(str => str.replace(/ /g, "_")); // Substitui espaços por underscores
      const ultimaLinha = buscarUltimaLinhaValida(aba, 1);
      const dados = aba.getRange(2, 1, ultimaLinha, aba.getLastColumn()).getValues();
      let nEntidade = 0;

      if (ultimaLinha !== -1) {
        for (let i = 0; i < ultimaLinha; i++) {
            //classe que representa uma linha
            if (dados[i][1] === "") 
                continue; // Pula linhas vazias na segunda coluna
            if (isNaN(parseFloat(dados[i][5])))
                continue; // Pula linhas onde a coluna f nao tem valor numérico
            if (dados[i][2] === "CTB")
                continue; // Pula linhas de curitiba
            objetoFinal[nomeAba][nEntidade] = {};
            for (let j = 0; j < categorias.length; j++) {
                //envia um vetor de linhas 
                objetoFinal[nomeAba][nEntidade][categorias[j]] = {};
                objetoFinal[nomeAba][nEntidade][categorias[j]] = dados[i][j];
            }
            //Contador de entidades, para que cada linha seja uma entidade diferente
            nEntidade++;
        }
      }
    })
    return ContentService.createTextOutput(JSON.stringify(objetoFinal))
    .setMimeType(ContentService.MimeType.JSON);
}

/**
 * retorna um vetor com os nomes de colunas da respectiva aba
 * @param {GoogleAppsScript.Spreadsheet.Sheet} aba 
 * @returns {string[]}
 */
function buscarCategorias(aba) {
  let vetor = [];
  const dados = aba.getRange(1, 1, 1, aba.getLastColumn()).getValues();

  for (let titulo of dados[0])
    if (titulo !== "") { 
      vetor.push(titulo);
    }

  return vetor;
}

/**
 * busca a primeira linha não vazia de baixo para cima na coluna especificada
 * @param {GoogleAppsScript.Spreadsheet.Sheet} aba 
 * @param {number} coluna 
 */
function buscarUltimaLinhaValida(aba, coluna) {
    const ultimaLinha = aba.getLastRow();
    const dadosColuna = aba.getRange(1, coluna, ultimaLinha).getValues();

    for (let i = ultimaLinha - 1; i >= 0; i--) 
        if (dadosColuna[i][0] !== "") 
            return i + 1; 
    
    return -1;
}
