/**
 * Função para buscar por nome na aba PSC
 * @param {object[][]} dadosPSC 
 * @param {object[][]} dadosPP 
 * @param {*} links 
 * @param {string} nome 
 * @returns {object}
 */
function buscarPorNome(dadosPSC, dadosPP, links, nome) {
  for (var i = 1; i < dadosPSC.length; i++) {
    //caso a caixa de email estiver marcada significa que está na fila para ser encerrado pelo acionador, portanto, não deve ser enviado
    const encerrado = dadosPSC[i][12] === 'Encerramento' || dadosPSC[i][12] === 'Indulto'; 
    if (dadosPSC[i][1] == nome && !encerrado) { // Coluna B (índice 1) é onde estão os nomes
      const ppCadastrada = !!dadosPSC[i][15]; // Coluna P (índice 15)
      let saldoPP = "";
      let penaValorPP = "";
      let dataSaldo = "";
      let parcelaPP = "";
      let contaPP = "";
      
      if (ppCadastrada) {
        const processo = dadosPSC[i][0];
        const ppDados = buscarDadosPP(processo);
        saldoPP = ppDados.saldo;
        penaValorPP = ppDados.penaValor;
        dataSaldo = ppDados.dataSaldo;
        parcelaPP = ppDados.parcela;
        contaPP = ppDados.conta;
      }
      return {
        processo: dadosPSC[i][0], // Coluna A
        nome: dadosPSC[i][1], 
        pena: dadosPSC[i][2], // Coluna C (índice 2)
        classe: dadosPSC [i][14],
        entidade: dadosPSC[i][7], // Coluna H (índice 7)
        executado: dadosPSC[i][5], // Coluna F (índice 5)
        relatorio: extrairLink(links[i - 1][0]), // Link correto da coluna K
        status: ppCadastrada ? "PP cadastrada" : "", // Coluna P (índice 15)
        juizo: dadosPSC[i][8], // Ajuste aqui para formatar corretamente
        saldo: saldoPP,
        penaPP: penaValorPP,
        dataPP: dataSaldo, 
        parcela: parcelaPP,
        conta: contaPP
      };
    }
  }
  for (var j = 1; j < dadosPP.length; j++) {
    if (dadosPP[j][1] == nome && dadosPP[j][19] !== 'Removido') { // Coluna B (índice 1) com o nome
      return {
        processo: dadosPP[j][0],
        nome: dadosPP[j][1],
        classe: dadosPP[j][20] || "",
        juizo: dadosPP[j][18] || "",
        saldo: dadosPP[j][3], // ajuste conforme coluna de saldo no PP
        penaPP: dadosPP[j][21] || "", // ajuste conforme coluna de pena no PP
        dataPP: dadosPP[j][24] || "",
        parcela: dadosPP[j][26] || "",
        conta: dadosPP[j][2] || ""
      };
    }
  }
  return { erro: "Nome não encontrado!" };
}

/**
 * Busca os dados referente ao processo na aba PP
 * @param {string} processo 
 * @returns {DadosPP}
 */
function buscarDadosPP(processo) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const abaPP = ss.getSheetByName("PP");
  const dados = abaPP.getDataRange().getValues();
  const dados2 = abaPP.getDataRange().getDisplayValues();
  for (let i = 1; i < dados.length; i++) {
    if (dados[i][0] == processo) {  // Coluna A = índice 0
      if (dados[i][19] === 'Removido')
        return {
          conta: "",
          saldo: "",
          penaValor: "",
          dataSaldo : "",
          parcela: ""
        };
      return {
        conta: dados[i][2], // coluna C
        saldo: dados[i][3],        // Coluna d 
        penaValor: dados[i][21] ,    // Coluna t = vira v
        dataSaldo : dados[i][24],  //coluna w - vira z
        parcela: dados2[i][26]// coluna AA
      };
    }
  }
  return {
    conta: "",
    saldo: "",
    penaValor: "",
    dataSaldo : "",
    parcela: ""
  };
}

/**
 * Função para extrair o link de uma célula com texto formatado
 * @param {GoogleAppsScript.Spreadsheet.RichTextValue} richText 
 * @returns {string} link
 */
function extrairLink(richText) {
  if (richText && richText.getRuns().length > 0) {
    var link = richText.getRuns()[0].getLinkUrl();
    return link ? link : "Sem link";
  }
  return "Sem link";
}

/**
 * retorna uma lista com as entidades para exibir no site ou
 * caso informe uma entidade, retorna a lista com seus prestadores
 * @param {e} e 
 * @param {GoogleAppsScript.Spreadsheet.Spreadsheet} planilha 
 * @return {object}
 */
function buscarEntidadesOuPrestadores(e, planilha) {
  var abaPSC = planilha.getSheetByName("PSC");
  var dadosPSC = abaPSC.getDataRange().getValues();

  var abaEntidades = planilha.getSheetByName("Entidades");
  var dadosEntidades = abaEntidades.getDataRange().getValues()

  // Se nenhum parâmetro "entidade" for enviado, retorna apenas os nomes das entidades
  if (!e.parameter.entidade || e.parameter.getEntidades !== undefined) {
    var entidadesSet = new Set();
    for (var i = 3; i < dadosEntidades.length; i++) {
      var entidade = dadosEntidades[i]; 
      if (entidade[0] != '' && ((entidade[1] != '' && entidade[1] != 'encerramento') || entidade[1] === 0) && entidade[0] !== 'Resultado:') {
        entidadesSet.add([entidade[0]]);
      }
    }
    var entidades = Array.from(entidadesSet);
    entidades.sort((a, b) => a[0].localeCompare(b[0], 'pt-BR', { sensitivity : 'base' }))
    return ContentService.createTextOutput(JSON.stringify({ entidades: entidades }))
           .setMimeType(ContentService.MimeType.JSON);
  }

  // Se uma entidade foi selecionada, retorna os prestadores dessa entidade.
  var entidadeSelecionada = e.parameter.entidade;
  var prestadores = [];
  for (var i = 1; i < dadosPSC.length; i++) {
    if (dadosPSC[i][7] === entidadeSelecionada) {
      prestadores.push(dadosPSC[i][1]); // Coluna 2 (índice 1)
    }
  }
  return ContentService.createTextOutput(JSON.stringify({ prestadores: prestadores }))
         .setMimeType(ContentService.MimeType.JSON);
}

/**
 * altera a celula entidade da psc ou a fiscalização da pp para indicar que 
 * o cadastro foi removido.
 * @param {GoogleAppsScript.Spreadsheet.Spreadsheet} planilha 
 * @param {string} prestador 
 * @param {boolean} induto 
 */
function excluirPena(planilha, prestador, induto) {
  const textoPP = 'Removido';
  const textoPSC = 'encerrado';
  
  if (induto === null) {
    const aba = planilha.getSheetByName('3PP');
    const colunaPrestador = buscarColuna(aba, 'PARTE - POLO PASSIVO');
    const coluna = buscarColuna(aba, 'Fiscalizar');
    const linha = buscarLinha(aba, colunaPrestador, prestador);
    if (linha != -1 && coluna != -1) {
      aba.getRange(linha, coluna).setValue(textoPP)
      console.log(`célula da linha ${linha} coluna ${coluna} atualizada`)
    }
  }
  else {
    const aba = planilha.getSheetByName('Lotação');
    const colunaPrestador = buscarColuna(aba, 'Prestador de serviços');
    const colunaTemporaria = buscarColuna(aba, 'Temporário');
    const linha = buscarLinha(aba, colunaPrestador, prestador);
    if (linha != -1 && colunaTemporaria != -1) {
      //insere na coluna temporaria para que o acionador encontre
      aba.getRange(linha, colunaTemporaria).setValue(textoPSC);
      const colunaEmail = buscarColuna(aba, 'E-mail');
      if (induto) 
        aba.getRange(linha, colunaEmail).setValue('Indulto');
      else
        aba.getRange(linha, colunaEmail).setValue('Encerramento');
      console.log(`célula da linha ${linha} coluna ${colunaTemporaria} atualizada, célula de email marcada para envio quando possível.`)
      SpreadsheetApp.flush();
    }
  }
}

/**

/**
 * busca uma coluna no cabeçalho
 * @param {GoogleAppsScript.Spreadsheet.Sheet} aba 
 * @param {string} nome 
 * @returns {number} coluna do excel ou -1
 */
function buscarColuna(aba, nome) {
  let dados = aba.getRange(1, 1, 1, aba.getLastColumn()).getValues()[0];
  let index = dados.indexOf(nome);
  return index != -1 ? index + 1 : -1 
}

/**
 * busca uma linha em uma coluna
 * @param {GoogleAppsScript.Spreadsheet.Sheet} aba 
 * @param {number} coluna 
 * @param {string} nome 
 * @returns {number} linha do excel ou -1
 */
function buscarLinha(aba, coluna, nome) {
  let dados = aba.getRange(1, coluna, aba.getLastRow(), coluna).getValues();
  let list = dados.map(dados => dados[0]);
  let index = list.indexOf(nome);
  return index != -1 ? index + 1 : -1;
}
