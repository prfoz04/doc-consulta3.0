/**
 * Porta Get única, decide o que vai ser retornado através dos parâmetros recebidos,
 * coleta os dados das planilhas .PSC e .PP.
 * @param {e} e 
 * @returns {*} retorna um JSON com as informações o front-end.
 */
function doGet(e) {
  try {

    //aba vagas do site
    if (e.parameter.vagas) {
      return handleVagasRequest(e);
    }
    //aba consulta do site
    if (e.parameter.aba || e.parameter.colunas) {
      return handlePPRequest(e);
    }
    //aba consulta do site
    if (e.parameter.entidades || e.parameter.getEntidades || 
        e.parameter.processo || e.parameter.nome) {
      return handlePSCRequest(e);
    }
    //aba execução de penas no site
    if (e.parameter.analiseDados) {
      return enviarDadosParaAnalise();
    }
    //aba destinação de valores no site
    if (e.parameter.destinacaoValores) {
      return destinacaoValores();
    }

    throw new Error("Requisição não reconhecida");
    
  } catch (err) {
    return ContentService.createTextOutput(
      JSON.stringify({error: err.message})
    ).setMimeType(ContentService.MimeType.JSON);
  }
}

/**
 * Porta única Post, insere e altera dados nas planilhas .PP e .PSC .
 * @param {e} e 
 * @returns {*} retorna um JSON de sucesso ou erro.
 */
function doPost(e) {
  Logger.log("Evento 'e' recebido: " + JSON.stringify(e)); 
  try {
    //insere um novo cadastro PP ou PSC
    if (e.parameter && (e.parameter.processo || e.parameter.nome)) {
      return handlePSCPost(e);
    }
    //altera cadastros já existentes
    if (e.parameter.excluirpp || e.parameter.excluirpsc || e.parameter.altPenapp || e.parameter.altPenapsc || e.parameter.altEntidade || e.parameter.altParcelas) {
      return atualizar(e.parameter);
    }

    const params = JSON.parse(e.postData.contents);
    
    if (params.mes) {
      return handlePPPost(e);
    }
    throw new Error("Tipo de POST não reconhecido");
    
  } catch (err) {
    return ContentService.createTextOutput(
      JSON.stringify({error: err.message})
    ).setMimeType(ContentService.MimeType.JSON);
  }
}

/**
 * função específica para busca na PP.
 * @param {e} e 
 * @returns {*} JSON
 */
function handlePPRequest(e) {
  var planilha = SpreadsheetApp.getActiveSpreadsheet();
  var nomeAba = e.parameter.aba || "";
  var colunas = e.parameter.colunas ? e.parameter.colunas.split(",") : [];

  if (!nomeAba || colunas.length === 0) {
    throw new Error("Especifique a aba e as colunas!");
  }

  var aba = planilha.getSheetByName(nomeAba);
  if (!aba) {
    throw new Error("Aba não encontrada!");
  }

  var dados = extrairColunas(aba, colunas);
  return ContentService.createTextOutput(JSON.stringify(dados))
    .setMimeType(ContentService.MimeType.JSON);

}

    range.setValues(dados);
    try{
      atualizarPP(mes, erros, data); // add data

    }
    catch{
      Logger.log("erro ao atualizar planilha de consulta PP");
    }

  } catch (erro) {
    return ContentService.createTextOutput(JSON.stringify({ erro: erro.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

/**
 * Função expecífica para buscas na .PSC.
 * @param {e} e 
 * @returns {*} JSON
 */
function handlePSCRequest(e) {
  var planilha = SpreadsheetApp.getActiveSpreadsheet();
  if (e.parameter.entidade !== undefined || e.parameter.getEntidades !== undefined) {
        return buscarEntidadesOuPrestadores(e, planilha);
  }
  var abaPSC = planilha.getSheetByName("PSC");
  var abaPP = planilha.getSheetByName("PP");
  var processoValor = e.parameter.processo || ""; 
  var nomeValor = e.parameter.nome || ""; 
  var dadosPSC = abaPSC.getDataRange().getValues(); // Obtém os valores da aba PSC
  var dadosPP = abaPP.getDataRange().getValues();
  var linksRelatorio = abaPSC.getRange(2, 11, dadosPSC.length - 1, 1).getRichTextValues(); // Obtém os links da coluna K (relatórios)
  var resultado = {};

  if (processoValor === "" && nomeValor !== "") {
    resultado = buscarPorNome(dadosPSC, dadosPP, linksRelatorio, nomeValor);

  } else if (nomeValor === "" && processoValor !== "") {
    resultado = buscarPorProcesso(dadosPSC, dadosPP, linksRelatorio, processoValor);

  } else {
    resultado.erro = "Insira um processo ou nome!";
  }
  if (resultado.executado === 'eproc' || resultado.executado == 'SEEU') {
    resultado['ignorado'] = true;
    resultado['motivo'] = `Informa as horas pelo ${resultado.executado}`;
  }
  return ContentService.createTextOutput(JSON.stringify(resultado))
    .setMimeType(ContentService.MimeType.JSON);
}

/**
 * busca a última linha preenchida em uma aba.
 * @param {GoogleAppsScript.Spreadsheet.Sheet} sheet 
 * @returns {number} numero da linha.
 */
function encontrarUltimaLinhaComConteudo(sheet) {
  const numLinhas = sheet.getMaxRows();
  const numColunas = sheet.getLastColumn();

  for (let i = numLinhas; i >= 1; i--) {
    const valores = sheet.getRange(i, 1, 1, numColunas).getValues()[0];
    const linhaTemDado = valores.some(cel => {
      if (typeof cel === 'string') {
        return cel.trim() !== '';
      }
      return cel !== '' && cel !== null;
    });
    if (linhaTemDado) {
      return i;
    }
  }
  return 1; // Caso nenhuma linha tenha dados
}

/**
 * Formata e insere os dados e fórmulas na aba 'Lotação' da planilha .PSC.
 * @param {object} dados
 */
function enviarParaLotacaoPSC(dados) {
    const urlDestino = "https://docs.google.com/spreadsheets/d/1OpwOjfYiqPXH0324DGbYDHf9syDlPcS0uIBB771BmmY/edit";
    const planilhaDestino = SpreadsheetApp.openByUrl(urlDestino);
    const abaLotacao = planilhaDestino.getSheetByName("Lotação");

    if (!abaLotacao) {
        throw new Error("Aba 'Lotação' não encontrada na planilha de destino.");
    }

    // Encontra a próxima linha vazia para inserção (adiciona 1 ao último conteúdo)
    let linhaDestino = encontrarUltimaLinhaComConteudo(abaLotacao) + 1;
    const totalNumColunas = 16; // Colunas A a P

    try {
        const novaLinha = new Array(totalNumColunas).fill(""); // Inicializa o array com strings vazias

        // Extrai e limpa a entidade uma vez
        const entidade = String(dados[7] || '').trim();

        // Preenche os valores diretos no array novaLinha (0-indexed)
        novaLinha[0] = dados[0];   // Coluna A: Processo
        novaLinha[1] = dados[1];   // Coluna B: Nome
        novaLinha[2] = dados[2];   // Coluna C: Pena
        novaLinha[3] = dados[3];   // Coluna D: Mínimo Mensal
        novaLinha[4] = dados[4];   // Coluna E: Máximo Mensal
        novaLinha[5] = `=IFS(C${linhaDestino}=\"\\\";\"\\\";C${linhaDestino}>0;IMPORTRANGE(L${linhaDestino}; CONCAT(B${linhaDestino};\"!$A$6\")))`; // Coluna F (Executado)
        if (entidade === "Conselho da Comunidade - Cascavel" || entidade === "Conselho da Comunidade - Pato Branco") { // Coluna G (A Cumprir) 
            novaLinha[6] = `\"SEEU\"`; // Texto literal
        } else {
            novaLinha[6] = `=IFS(C${linhaDestino}=\"\\\";\"\\\";C${linhaDestino}>0;C${linhaDestino}-F${linhaDestino})`;
        }
        novaLinha[7] = entidade;   // Coluna H: Entidade
        novaLinha[8] = dados[8];   // Coluna I: Juízo
        novaLinha[9] = `=IF(ISERROR(F${linhaDestino}*100/C${linhaDestino});\"\\\";F${linhaDestino}*100/C${linhaDestino})`; // Coluna J (%)
        novaLinha[10] = `=IFS(C${linhaDestino}=\"\\\";\"\\\";C${linhaDestino}>0;IMPORTRANGE(L${linhaDestino}; CONCAT(B${linhaDestino};\"!$A$7\")))`; // Coluna K (Relatório)
        novaLinha[11] = `=IF(H${linhaDestino}=\"\\\";\"\\\";VLOOKUP(H${linhaDestino};\'Quadro de Vagas\'!C:D;2;FALSE))`; // Coluna L (Links)
        novaLinha[12] = "";        // Coluna M: E-mail (permanece vazio)
        novaLinha[13] = obterOpcaoMesAtual(); // Coluna N: Fiscalização
        novaLinha[14] = dados[14]; // Coluna O: Classes
        novaLinha[15] = `=IF(A${linhaDestino}=\"\\\";\"\\\";ISTEXT(VLOOKUP(A${linhaDestino};IMPORTRANGE(\"https://docs.google.com/spreadsheets/d/1ouGkmzF8_dJUZPXRigsL9qpgZ9hMiBOtaqbZs3utxNU/edit?gid=1597838336\";\"3PP!A:O\");1;FALSE)))`; // Coluna P (PP)
        
        //MUDANÇA -- envia tudo de uma vez para melhorar desempenho
        abaLotacao.getRange(linhaDestino, 1, 1, totalNumColunas).setValues([novaLinha])

        Logger.log("Dados adicionados à aba 'Lotação' na linha " + linhaDestino);
    } catch (erro) {
        Logger.log("Erro ao enviar para Lotação: " + erro);
        // Limpa a linha se um erro ocorrer durante a inserção
        if (linhaDestino <= abaLotacao.getMaxRows() + 1) { // Verifica se a linha foi adicionada ou existia
            abaLotacao.getRange(linhaDestino, 1, 1, totalNumColunas).clearContent();
        }
        throw erro; // Re-lança o erro para ser tratado por handlePSCPost
    }
}

/**
 * Formata e insere os dados na primeira aba da planilha .PP.
 * @param {object} dados 
 */
function enviarParaLotacaoPP(dados) {
  const urlDestino = "https://docs.google.com/spreadsheets/d/1ouGkmzF8_dJUZPXRigsL9qpgZ9hMiBOtaqbZs3utxNU/edit";
  const planilhaDestino = SpreadsheetApp.openByUrl(urlDestino);
  const abaPP = planilhaDestino.getSheetByName("3PP");
  if (!abaPP) throw new Error("Aba '3PP' não encontrada.");

  const NUM_COLUNAS = 28;
  let linhaDestino = proximaLinhaInsercao(abaPP);;

  try {
    const novaLinha = new Array(28).fill("");
    const colunasParaPreencher = {
      1: dados[0],    // A: Processo
      2: dados[1],    // B: Nome
      19: dados[8],   // Q: Juízo - vira S
      21: dados[14],  // S: Classe vira U
      22: dados[16],  // T: PenaPP vira v
      23: dados[17],  // U: Parcela vira w
    };
    for (const [coluna, valor] of Object.entries(colunasParaPreencher)) {
      novaLinha[coluna - 1] = valor; // Arrays começam em 0
    }
    novaLinha[19] = obterOpcaoMesAtual(); // Coluna T
    abaPP.getRange(linhaDestino, 1, 1, 28).setValues([novaLinha]);

    const formulaZ = `=NOT(ISNA(VLOOKUP(A${linhaDestino};IMPORTRANGE(\"https://docs.google.com/spreadsheets/d/1OpwOjfYiqPXH0324DGbYDHf9syDlPcS0uIBB771BmmY/edit\";\"Lotação!A2:C\");3;FALSE)))`;
    abaPP.getRange(linhaDestino, 18).setFormula(formulaZ);
    // Linha já existente (coluna R = 18)

    //fórmula para coluna X (coluna 24)
    const formulaX = `=IF(ISNUMBER(V${linhaDestino});D${linhaDestino}/V${linhaDestino};"")`;
    abaPP.getRange(linhaDestino, 24).setFormula(formulaX);

    // fórmula para coluna Z (coluna 26)
    const formulaZ2 = `=IFERROR(IFS($Z$1="Janeiro";E${linhaDestino}/D${linhaDestino};
                                    $Z$1="Fevereiro";F${linhaDestino}/D${linhaDestino};
                                    $Z$1="Março";G${linhaDestino}/D${linhaDestino};
                                    $Z$1="Abril";H${linhaDestino}/D${linhaDestino};
                                    $Z$1="Maio";I${linhaDestino}/D${linhaDestino};
                                    $Z$1="Junho";J${linhaDestino}/D${linhaDestino};
                                    $Z$1="Julho";K${linhaDestino}/D${linhaDestino};
                                    $Z$1="Agosto";L${linhaDestino}/D${linhaDestino};
                                    $Z$1="Setembro";M${linhaDestino}/D${linhaDestino};
                                    $Z$1="Outubro";N${linhaDestino}/D${linhaDestino};
                                    $Z$1="Novembro";O${linhaDestino}/D${linhaDestino};
                                    $Z$1="Dezembro";P${linhaDestino}/D${linhaDestino});N(0))`;
    abaPP.getRange(linhaDestino, 26).setFormula(formulaZ2);

    const formulaAA = `=IF(ISERROR(D${linhaDestino}/(V${linhaDestino}/W${linhaDestino}));"";IF(D${linhaDestino}/(V${linhaDestino}/W${linhaDestino})<W${linhaDestino};CONCATENATE(INT(D${linhaDestino}/(V${linhaDestino}/W${linhaDestino}));"/";W${linhaDestino});"ERRO"))`;
    abaPP.getRange(linhaDestino, 27).setFormula(formulaAA); // Coluna AA
    
    const formulaAB = `=IF(COUNTIF(A:A;A${linhaDestino})>1;"Duplicado";"")`;
    abaPP.getRange(linhaDestino, 28).setFormula(formulaAB); // Coluna AB

    Logger.log(`Dados adicionados na linha ${linhaDestino}`);
  } catch (e) {
    
    abaPP.getRange(linhaDestino, 1, 1, NUM_COLUNAS).clearContent();
    
    Logger.log("Erro ao enviarParaLotacaoPP: " + e);
    throw e;
  }
}

/**
 * função que decide quando haverá fiscalização baseado no mês de cadastro,
 * como ocorre mês sim e mês não tem as opções 'ímpar' e 'par'.
 * @returns {string} opção de fiscalização selecionada.
 */
function obterOpcaoMesAtual() {
  const opcoesSelecao = ["PAR", "ÍMPAR"];
  const dataAtual = new Date();
  const mesAtual = dataAtual.getMonth() + 1; //pois a função retorna janeiro 0 
  return opcoesSelecao[mesAtual % 2]; //nova lógica de fiscalização a cada dois meses
}

/**
 * Busca a próxima linha real onde deve ser inserido um dado na aba,
 * caso não exista linha vazia, cria uma
 * @param {GoogleAppsScript.Spreadsheet.Sheet} abaPP 
 * @returns {number} número da linha encontrada ou criada
 */
function proximaLinhaInsercao(abaPP) {
  const NUM_COLUNAS = 2;

  const ultimaLinha = abaPP.getLastRow();
  let linha = ultimaLinha;

  while (linha >= 1) {
    const valores = abaPP.getRange(linha, 1, 1, NUM_COLUNAS).getDisplayValues()[0];
    const temConteudo = valores.some(v => v !== '' && v !== null);
    if (temConteudo) break;
    linha--;
  }

  const linhaDestino = linha + 1;
  const totalLinhas = abaPP.getMaxRows();

  if (linhaDestino > totalLinhas) {
    abaPP.insertRowsAfter(totalLinhas, 1);
  }

  return linhaDestino;
}

/**
 * Retorna as informações referente a entidades obtida na aba 'vagas', informações essas que são um import
 * da aba 'Quadro de Vagas' da planilha .PSC.
 * @param {e} e 
 * @returns {*} JSON
 */
function handleVagasRequest(e) {
  const ENTIDADE_DELIMITADORA = "x PSC encerrada";
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("vagas");

  if (!sheet) {
    throw new Error(`A aba vagas não foi encontrada.`);
  }
  const allData = sheet.getDataRange().getValues();
  const vagasFiltradas = [];

  for (let i = 2; i < allData.length; i++) {
    const row = allData[i];
    const nomeEntidade = String(row[0] || '').trim();
    const textoPadrao = row[3];
    const local = row[4];
    const vagasDisponiveis = row[8];
    const observacao = row[13];         

    if (nomeEntidade.toLowerCase() === ENTIDADE_DELIMITADORA.toLowerCase()) {
      Logger.log(`Leitura interrompida na entidade: ${nomeEntidade}`);
      break; // INTERROMPE o loop
    }
    if (local.toLowerCase() === "diversas") {
        continue;
    }
    const statusVaga = String(vagasDisponiveis).trim().toLowerCase();

    if (statusVaga !== "encerramento" && nomeEntidade) {
      vagasFiltradas.push({
        entidade: nomeEntidade,
        texto: textoPadrao,
        local: local,
        vagas: vagasDisponiveis,
        obs:  observacao
      });
    }
  }

  const jsonResponse = {
    count: vagasFiltradas.length,
    data: vagasFiltradas
  };

  return ContentService.createTextOutput(
    JSON.stringify(jsonResponse)
  ).setMimeType(ContentService.MimeType.JSON);
}

/**
 * realiza alterações em cadastros nas planilhas PP e PSC
 * @param {parametros} params 
 * @returns retorna o resultado positivo ou negativo para o front
 */
function atualizar(params) {
  try {
    //planilhas
    const PSC = SpreadsheetApp.openByUrl("https://docs.google.com/spreadsheets/d/1OpwOjfYiqPXH0324DGbYDHf9syDlPcS0uIBB771BmmY/edit");
    const PP = SpreadsheetApp.openByUrl("https://docs.google.com/spreadsheets/d/1ouGkmzF8_dJUZPXRigsL9qpgZ9hMiBOtaqbZs3utxNU/edit");

    // Converte os parâmetros de string para booleano 
    const excluirpp = params.excluirpp === "true";
    const excluirpsc = params.excluirpsc === "true";
    const induto = params.induto === "true";
    const altEntidade = params.altEntidade === "true";
    const altPenapp = params.altPenapp === "true";
    const altPenapsc = params.altPenapsc === "true";
    const altParcelas = params.altParcelas === "true";

    if (excluirpp) {
      excluirPena(PP, params.prestador, null);
    }
    if (excluirpsc) {
      excluirPena(PSC, params.prestador, induto);
    }
    if (altEntidade) {
      alterarEntidade(PSC, params.prestador, params.novaEntidade);
    }
    if (altPenapp) {
      alterarPena(PP, params.prestador, params.penapp);
    }
    if (altPenapsc) {
      alterarPena(PSC, params.prestador, params.penapsc);
    }
    if (altParcelas) {
      alterarParcela(PP, params.prestador, params.parcelas);
    }
    return ContentService.createTextOutput(JSON.stringify({ sucesso: "Dados atualizados com sucesso!" }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (erro) {
    return ContentService.createTextOutput(JSON.stringify({ erro: erro.message }))
    .setMimeType(ContentService.MimeType.JSON);
  }
}
