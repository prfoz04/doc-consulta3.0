function consultar() {
  var planilha = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Consulta");
  var processo = planilha.getRange("D1");
  var nome = planilha.getRange("D2");

  // Obtém os valores das células
  var processoValor = processo.getValue();
  var nomeValor = nome.getValue();

  // Define os valores e fórmulas dos intervalos
  var intervaloValores = planilha.getRange("C5:C9");

  var intervaloFormulas = planilha.getRange("D5:D8");

  if (processoValor === "" && nomeValor !== "") {
    planilha.getRange("D1").setFormula('=INDEX(PSC!A:A;MATCH(D2;PSC!B:B;0))');
    intervaloValores.setValues([
    ['Pena'],
    ['Entidade'],
    ['Executado'],
    ['Relatório'],
    ['=IF(PROCV_PSC(16);"PP cadastrada";"")']
  ]);
    intervaloFormulas.setFormulas([
    ['=PROCV_PSC(3)'],
    ['=PROCV_PSC(8)'],
    ['=PROCV_PSC(6)'],
    ['=PROCV_PSC(11)']
  ]);
  } else if (nomeValor === "" && processoValor !== "") {
    planilha.getRange("D2").setFormula('=PROCV_PSC(2)');
    intervaloValores.setValues([
    ['Pena'],
    ['Entidade'],
    ['Executado'],
    ['Relatório'],
    ['=IF(PROCV_PSC(16);"PP cadastrada";"")']
  ]);
    intervaloFormulas.setFormulas([
    ['=PROCV_PSC(3)'],
    ['=PROCV_PSC(8)'],
    ['=PROCV_PSC(6)'],
    ['=PROCV_PSC(11)']
  ]);
  } else if (nomeValor === "" && processoValor === "") {
    SpreadsheetApp.getUi().alert('Insira processo ou nome!');
  }
}
function limparCelulas() {
  // Obtém a planilha com o nome "consulta"
  var planilha = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Consulta");
  planilha.getRange("D1:I2").clearContent();
  planilha.getRange("C5:D10").clearContent();
